  attributes.swift
  children.swift
  constants.swift
  copy.swift
  currentWorkingDirectory.swift
  FilePermission.swift
  FileTime.swift
  FileType.swift
  glob.swift
  helpers.swift
  manipulate.swift

  pathAnalysis.swift
  pathAnalysisWithIO.swift
  PathosError.swift
  PathRepresentable.swift

  posixHelpers.swift

  $<$<BOOL:${BUILD_TESTING}>:-enable-testing>)
